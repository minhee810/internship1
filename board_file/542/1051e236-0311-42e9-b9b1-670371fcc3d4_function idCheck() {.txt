function idCheck() {
			let username = document.getElementById("username");
			let idFeedbackSpan = document.getElementById("id_feedback");
			if (username.value.length < 4 || username.value.length > 11) {
				idFeedbackSpan.innerHTML = "<font color=red>아이디는 4자 이상, 10자 이하여야 합니다. </font>";
			} else {
				let xhr = new XMLHttpRequest();
				xhr.onreadystatechange = function() {
					if (xhr.readyState == 4 && xhr.status == 200) {
						idFeedbackSpan.innerHTML = "<font color=blue> 사용 가능한 아이디입니다. </font>";
					} else {
						idFeedbackSpan.innerHTML = "<font color=blue> 이미 사용중인 아이디입니다. </font>";
					}
				}
			}
			xhr.open("get", "idCheck?id=" + username.value);
			xhr.send();
		}
